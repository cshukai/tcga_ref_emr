
# display all the url
prefix tcga:<http://purl.org/tcga/core#>

select distinct ?url 

where {

  ?file tcga:url ?url .
}


# display all the diseases

prefix tcga:<http://purl.org/tcga/core#>

select  distinct ?diseaseType

where {

  ?file tcga:disease-study ?diseases .
  ?diseases rdfs:label ?diseaseType.  
}

# display  names of  image files of gbm patients
prefix tcga:<http://purl.org/tcga/core#>

select distinct ?dataType ?name ?url where {
  
  ?file tcga:url ?url .
  ?file rdfs:label ?name .
  filter ( contains(?name, '.svs') ) .
  ?file tcga:data-type ?d.
  ?d rdfs:label ?dataType.
  ?file tcga:disease-study ?ds.
  ?ds  rdfs:label 'gbm'.
}
 
# display names of  clinical files of gbm patients
prefix tcga:<http://purl.org/tcga/core#>

select distinct ?url where {
  ?file tcga:url ?url .
  filter ( contains(?url, 'gbm') ) .
   filter ( contains(?url, 'clin') ) .
  filter ( contains(?url, '.txt') ) .
  filter ( contains(?url, 'public') ) .
  
}
